import {MAINNET} from "unchained-bitcoin";
import {TrezorExportExtendedPublicKey} from "unchained-wallets";
const interaction = new TrezorExportExtendedPublicKey({network: MAINNET, bip32Path: "m/48'/0'/0'"});
